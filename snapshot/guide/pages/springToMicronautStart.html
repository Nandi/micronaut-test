2.1 Build Configuration

Version: 1.0.0.BUILD-SNAPSHOT

2.1 Build Configuration

Micronaut 1.0.1 or above is required and the following is a minimal build to get this working for Spring (a non-web application):

plugins {
    id "java"
    id "org.springframework.boot" version "2.1.0.RELEASE"
    id "io.spring.dependency-management" version "1.0.6.RELEASE"
    id "net.ltgt.apt" version "0.19"
}

dependencies {
    annotationProcessor "io.micronaut:micronaut-inject-java:1.0.1"
    annotationProcessor "io.micronaut:micronaut-validation::1.0.1"
    annotationProcessor "io.micronaut.spring:micronaut-spring-annotation:1.0.0.BUILD-SNAPSHOT"

    compile "io.micronaut:micronaut-inject::1.0.1"
    compile("org.springframework.boot:spring-boot-starter")
    ...

}
If you are only going to references Spring annotations and not Spring classes or interfaces then org.springframework.boot:spring-boot-starter can be compileOnly
